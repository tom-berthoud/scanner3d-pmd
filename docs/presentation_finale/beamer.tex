% SLIDE 2 : LE PROJET & LE CAHIER DES CHARGES  (Tom -- ~50 s)
% ============================================================
\begin{frame}{Le projet}{Pourquoi un scanner 3D, et pour qui ?}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.56\textwidth}
      \begin{itemize}
        \item \textbf{Le besoin :} num\'{e}riser un objet physique pour obtenir un fichier 3D (\textbf{STL/OBJ}) directement imprimable.
        \item \textbf{Pour le MakeLab} de la HEIG-VD : un outil simple, autonome et open-source.
        \item \textbf{Le principe :} l'objet tourne sous une ligne laser, une cam\'{e}ra filme, le logiciel reconstruit le volume.
      \end{itemize}

      \vspace{0.3cm}
      \footnotesize\textbf{Cahier des charges (extraits)}
      \begin{tabularx}{\linewidth}{@{}>{\bfseries}p{2.5cm}X@{}}
        \toprule
        Objets    & jusqu'\`{a} $150\times150\times150$\,mm \\
        Dur\'{e}e     & un scan en $< 10$ min \\
        Pr\'{e}cision & cible $\leq 2$ mm \\
        Budget    & $\leq 300$ CHF \\
        Sortie    & fichier STL / OBJ \\
        \bottomrule
      \end{tabularx}
    \end{column}
    \begin{column}{0.42\textwidth}
      \hspace{0.4cm}
      \includegraphics[width=\linewidth,height=5.8cm,keepaspectratio]{vue_ext_ferme.jpeg}
    \end{column}
  \end{columns}

  \speaker{Tom}{Chef de projet}
\end{frame}

% ============================================================
% SLIDE 3 : PRINCIPE DE TRIANGULATION  (Tom -- ~40 s)
% ============================================================
\begin{frame}{Comment \c{c}a marche}{La triangulation laser}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{enumerate}
        \item Le plateau tourne pas \`{a} pas ($200 \times 1{,}8^\circ$).
        \item Le laser ligne dessine un profil de l'objet.
        \item La cam\'{e}ra voit la d\'{e}formation de la ligne.
        \item Par g\'{e}om\'{e}trie : chaque pixel $\rightarrow$ un point 3D.
        \item On empile les profils sur $360^\circ$ $\rightarrow$ nuage de points $\rightarrow$ maillage STL.
      \end{enumerate}
    \end{column}
    \begin{column}{0.50\textwidth}
      \hspace{0.4cm}
      \input{schemas/triangulation.tex}
    \end{column}
  \end{columns}

  \speaker{Tom}{Du r\'{e}el au mod\`{e}le 3D}
\end{frame}

% ============================================================
% SLIDE 6 : LOGICIEL -- MONO CAMERA & OCCULTATION  (Hadrien -- ~50 s)
% ============================================================
\begin{frame}{Le logiciel}{Une cam\'{e}ra : \c{c}a marche... presque}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.50\textwidth}
      \footnotesize
      \textbf{Pipeline :} Acquisition $\rightarrow$ Extraction de la ligne $\rightarrow$ Triangulation $\rightarrow$ Reconstruction $\rightarrow$ Export.

      \vspace{0.25cm}
      \begin{itemize}
        \item Formes \textbf{rondes} : r\'{e}sultat propre et exploitable.
        \item Formes \textbf{anguleuses} : le laser touche la pi\`{e}ce, mais une face cache le point \`{a} la cam\'{e}ra.
        \item R\'{e}sultat : des \textbf{trous} dans le mod\`{e}le, impossibles \`{a} corriger en logiciel.
      \end{itemize}

      \vspace{0.1cm}
      \limitbox{Occultation : limite physique d'un montage \`{a} une seule cam\'{e}ra.}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \input{schemas/occultation.tex}\\[0.1cm]
      \scriptsize Zone \'{e}clair\'{e}e mais invisible pour la cam\'{e}ra
    \end{column}
  \end{columns}

  \speaker{Hadrien}{Acquisition \& traitement -- 1\textsuperscript{re} cam\'{e}ra}
\end{frame}
