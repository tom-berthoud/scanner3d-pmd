\begin{frame}{Principe de triangulation}{Du laser au mod\`{e}le 3D}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.46\textwidth}
      \begin{enumerate}
        \item Le plateau tourne pas \`{a} pas ($200\times1{,}8^\circ$).
        \item Le laser ligne \'{e}claire un profil de l'objet.
        \item La cam\'{e}ra capture la d\'{e}formation de la ligne.
        \item Triangulation : pixel $\rightarrow$ point 3D.
        \item Fusion des profils sur $360^\circ$ $\rightarrow$ nuage de points $\rightarrow$ STL.
      \end{enumerate}

      \vspace{0.15cm}
      \footnotesize
      \textbf{Pipeline :} Acquisition $\rightarrow$ Extraction $\rightarrow$ Triangulation $\rightarrow$ Reconstruction $\rightarrow$ Export

% SLIDE 6b : PIPELINE LOGICIEL ET NOUVEAUX ALGORITHMES
% ============================================================
\begin{frame}{Cha\^{i}ne logicielle}{Les traitements ajout\'{e}s depuis la mi-projet}
  \footnotesize
  \begin{enumerate}
    \item \textbf{Calibration d'exposition} (avant scan) -- largeur de ligne laser cibl\'{e}e par cam\'{e}ra.
    \item \textbf{Acquisition} -- pas moteur, laser, capture des deux cam\'{e}ras.
    \item \textbf{Extraction + triangulation} -- pixels laser $\rightarrow$ points 3D (mm).
    \item \textbf{Fusion par pas} -- les deux cam\'{e}ras combin\'{e}es en un profil lisse.
    \item \textbf{Fusion demi-tour} -- d\'{e}duplication des points revus \`{a} $180^\circ$.
    \item \textbf{Nettoyage} -- plan sup\'{e}rieur d\'{e}tect\'{e}, artefacts au-dessus rogn\'{e}s.
    \item \textbf{Reconstruction + export} -- filtrage k-NN, maillage, STL/OBJ.
  \end{enumerate}

  \vspace{0.15cm}
  \begin{center}
    \tagbox{softblue}{Nouveaut\'{e}s}\quad \'{E}tapes 1, 4, 5 et 6 ajout\'{e}es pour fiabiliser le nuage de la configuration double cam\'{e}ra.
  \end{center}
\end{frame}

% ============================================================
% SLIDE 6c : CALIBRATION D'EXPOSITION AUTOMATIQUE
% ============================================================
\begin{frame}{Calibration d'exposition automatique}{\texttt{orchestration/scan.py} -- avant chaque scan}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.54\textwidth}
      \footnotesize
      \begin{itemize}
        \item Pour chaque cam\'{e}ra, on vise une \textbf{largeur m\'{e}diane} de ligne laser (\texttt{target\_median\_width\_px}).
        \item On capture quelques \'{e}chantillons \`{a} diff\'{e}rents angles et on mesure la largeur r\'{e}elle.
        \item Ajustement multiplicatif de l'exposition :
          \[ t_{n+1} = t_n \cdot \frac{w_{\text{cible}}}{w_{\text{mesur\'{e}}}} \]
        \item Facteur born\'{e} (\texttt{max\_adjust\_factor}), exposition born\'{e}e (min/max).
        \item It\'{e}ration jusqu'\`{a} $|w - w_{\text{cible}}| \leq$ \texttt{tolerance\_width\_px} ou \texttt{max\_iterations}.
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \small
      \textbf{Pourquoi ?}
      \begin{itemize}
        \item Une ligne trop fine $\rightarrow$ points manquants.
        \item Une ligne satur\'{e}e $\rightarrow$ centro\"{i}de impr\'{e}cis.
      \end{itemize}
      \vspace{0.15cm}
      \validbox{Largeur de ligne homog\`{e}ne entre les deux cam\'{e}ras avant la capture.}
    \end{column}
  \end{columns}
\end{frame}

% ============================================================
% SLIDE 6d : FUSION DES PROFILS
% ============================================================
\begin{frame}{Fusion des profils}{\texttt{reconstruction/pointcloud.py} -- deux m\'{e}canismes}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \small\textbf{\textcolor{deepblue}{Fusion par pas (inter-cam\'{e}ra)}}\\[0.1cm]
      \footnotesize
      \texttt{\_fuse\_step\_profiles}
      \begin{itemize}
        \item Axe principal du profil par \textbf{ACP}, points tri\'{e}s le long de cet axe.
        \item D\'{e}coupe en segments selon les \'{e}carts (\texttt{gap\_mm}).
        \item Sur les zones vues par les \textbf{deux} cam\'{e}ras : lissage polynomial local glissant.
        \item Une seule courbe coh\'{e}rente au lieu de deux nuages superpos\'{e}s.
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \small\textbf{\textcolor{lasergreen}{Fusion demi-tour}}\\[0.1cm]
      \footnotesize
      \texttt{fuse\_half\_turn\_profiles}
      \begin{itemize}
        \item Un point du dessus est revu vers le pas $i + n/2$.
        \item Appariement par \textbf{plus proches voisins mutuels} (distance xyz).
        \item Paires sous \texttt{max\_pair\_distance\_mm} moyenn\'{e}es ; points isol\'{e}s conserv\'{e}s.
        \item R\'{e}duit doublons et bruit sur les faces sup\'{e}rieures.
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.1cm}
  \begin{center}
    \tagbox{softblue}{Effet}\quad Nuage plus dense et plus r\'{e}gulier, moins de d\'{e}doublement de surfaces.
  \end{center}
\end{frame}

% ============================================================
% SLIDE 6e : NETTOYAGE PLAN SUPERIEUR
% ============================================================
\begin{frame}{Nettoyage du nuage : plan sup\'{e}rieur}{\texttt{detect\_top\_plane} / \texttt{clip\_above\_detected\_top\_plane}}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.56\textwidth}
      \footnotesize
      \begin{itemize}
        \item Balayage de bandes horizontales du nuage de r\'{e}f\'{e}rence (haut vers bas).
        \item On retient la bande la plus haute qui soit \textbf{dense}, \textbf{\'{e}tendue} et \textbf{plane}.
        \item Plan ajust\'{e} par \textbf{SVD}, normale orient\'{e}e vers $+Y$ (le haut).
        \item Tout ce qui d\'{e}passe (au-del\`{a} de \texttt{clip\_margin\_mm}) est rogn\'{e}.
        \item Option : capuchons plats pour refermer le dessus.
      \end{itemize}
    \end{column}
    \begin{column}{0.40\textwidth}
      \small
      \textbf{Probl\`{e}me trait\'{e} :}
      \begin{itemize}
        \item Artefacts \og en pointe \fg{} au-dessus de l'objet.
        \item Sommet bruit\'{e} par la fusion demi-tour.
      \end{itemize}
      \vspace{0.15cm}
      \validbox{Dessus de pi\`{e}ce net, sans points flottants parasites.}
    \end{column}
  \end{columns}
\end{frame}
